--PRICE UP VOLUME UP
select sm.NSE_SYMBOL, sp.CURRENT_PRICE, st.ROCV from STOCK_MASTER sm, STOCK_PRICE sp, STOCK_TECHNICALS st 
where sm.STOCK_ID=sp.STOCK_ID
and sm.STOCK_ID = st.STOCK_ID
and sp.CURRENT_PRICE > sp.PREV_CLOSE
and st.ROCV > 50.0
order by st.ROCV desc

--PRICE DOWN VOLUME UP
select sm.NSE_SYMBOL, sp.CURRENT_PRICE, st.ROCV from STOCK_MASTER sm, STOCK_PRICE sp, STOCK_TECHNICALS st 
where sm.STOCK_ID=sp.STOCK_ID
and sm.STOCK_ID = st.STOCK_ID
and sp.CURRENT_PRICE < sp.PREV_CLOSE
and st.ROCV > 50.0
order by st.ROCV desc

AND bl.TYPE='POSITIVE'

AND bl.TYPE='POSITIVE'
